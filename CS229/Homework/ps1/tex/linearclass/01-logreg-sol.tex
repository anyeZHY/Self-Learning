\begin{answer}
	\[
		\nabla J
		=
		-\frac{1}{n}\sum^{n}_{i=1} \qty[\qty(y^{(i)}-g(\theta^Tx^{(i)}))x^{(i)}]
	.\] 
	Then
	\[
	    H=\nabla^2 J
		=
		\frac{1}{n}\sum^{n}_{i=1} 
		\qty{g(\theta^Tx^{(i)})\qty[1-g(\theta^Tx^{(i)})]x^{(i)}x^{(i)T}}
	.\] 
	By hint above we have $H\succeq 0$.
\end{answer}
