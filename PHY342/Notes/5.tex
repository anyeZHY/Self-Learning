\subsection{Probability current}

\begin{defi}[Probability current]
	\[
	    J
		\defeq
		-\frac{i\hbar}{2m}(\Psi^*\nabla\Psi-\Psi\nabla\Psi^*)
		=
		\frac{\hbar}{m}\Im(\Psi^*\nabla\Psi)
	.\] 
\end{defi}

Then 
\[
	\pdv{\rho}{t}+\nabla J = 0
\] 
where $\displaystyle \rho=\int\Psi^*\Psi\,\dd{x}$.
